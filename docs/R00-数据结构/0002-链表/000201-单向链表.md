# 单向链表

```java
package com.rzcode.ds.linkedlist;

import java.util.*;

public class SinglyLinkedList {
    /**
     * 链表的头节点
     */
    private Node head;

    /**
     * 添加节点到链表的末尾
     *
     * @param value
     */
    public void add(int value) {
        Node newNode = new Node(value);
        if (head == null) {
            head = newNode;
        } else {
            Node current = head;// 从头开始遍历
            while (current.next != null) {
                current = current.next;
            }
            current.next = newNode;
        }

    }

    // 在指定位置插入节点
    public void addAtPosition(int position, int value) {
        Node newNode = new Node(value);// 创建新节点
        if (position == 0) {
            newNode.next = head;//新节点的下一个节点指向当前头节点
            head = newNode;//更新头节点为新节点
            return;
        }
        Node current = head;
        for (int i = 0; i < position - 1; i++) {
            if (current == null) {
                throw new IndexOutOfBoundsException("Position out of bounds"); // 位置超出链表长度
            }
            current = current.next; // 移动到下一个节点
        }

        newNode.next = current.next; // 新节点的下一个节点指向当前节点的下一个节点
        current.next = newNode;       // 当前节点的下一个节点指向新节点
    }

    /**
     * 删除指定的节点
     *
     * @param value
     */
    public void delete(int value) {
        // 处理空链表的情况
        if (head == null) {
            return;
        }
        // 创建一个虚拟头节点，方便处理头节点的删除
        Node dummy = new Node(0);// 假设Node的构造函数接受一个值
        dummy.next = head;
        Node current = dummy; //从虚拟头节点开始
        while (current.next != null) {
            if (current.next.value == value) {
                current.next = current.next.next;// 删除节点
            } else {
                current = current.next;// 移动到下一个节点
            }
        }
        // 更新头节点
        head = dummy.next;
    }

    // 查找节点
    public boolean contains(int value) {
        if (head.value == value) {
            return true;
        }
        Node current = head;
        while (current.next != null) {
            if (current.next.value == value) {
                return true;
            }
        }
        return false;
    }

    public Node get(int value) {
        Node current = head; // 从头节点开始
        while (current != null) { // 遍历整个链表
            if (current.value == value) { // 如果找到匹配的值
                return current; // 返回当前节点
            }
            current = current.next; // 移动到下一个节点
        }
        return null; // 如果没有找到，返回 null
    }

    public List<Node> getAll(int value) {
        List<Node> matchingNodes = new ArrayList<>();//存储所有匹配的节点
        Node current = head;// 从头节点开始
        while (current != null) {
            if (current.value == value) {
                matchingNodes.add(current); // 将匹配的节点添加到列表中
            }
            current = current.next;
        }
        return matchingNodes;
    }

    /**
     * 显示链表中的所有节点
     */
    public void display() {
        Node current = head; // 从头节点开始
        while (current != null) {
            System.out.print(current.value + " -> "); // 打印节点值，不换行
            current = current.next; // 移动到下一个节点
        }
        System.out.println("null"); // 最后输出 null，表示链表结束
    }

    /**
     * 内部节点
     */
    class Node {
        /**
         * 节点存储的值
         */
        int value;

        /**
         * 指向下一个节点的引用
         */
        Node next;

        Node(int value) {
            // 初始化节点值
            this.value = value;
            // 默认下一个节点为null
            this.next = null;
        }
    }

    public static void main(String[] args) {
        SinglyLinkedList list = new SinglyLinkedList();
        list.add(1);
        list.add(3);
        list.display();
    }

}
```
