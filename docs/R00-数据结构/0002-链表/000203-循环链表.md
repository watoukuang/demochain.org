# 循环链表
```java
package com.rzcode.ds.linkedlist;

public class CircularLinkedList {

    class Node {
        int value;
        Node next;

        public Node(int value) {
            this.value = value;
            this.next = null;
        }
    }

    private Node tail;


    public CircularLinkedList() {
        this.tail = null;
    }

    /**
     * 插入链表到循环链表的末尾
     */
    private void insertAtEnd(int value) {
        Node newNode = new Node(value);
        if (tail == null) {
            tail = newNode;
            tail.next = tail;// 指向自身，形成循环
        } else {
            newNode.next = tail.next;// 新节点指向头节点
            tail.next = newNode;// 当前尾节点执行新节点
            tail = newNode; // 更新节点
        }
    }

    /**
     * 在循环链表的头部插入节点
     */
    public void insertAtHead(int value) {
        Node newNode = new Node(3);
        if (tail == null) {
            tail = newNode;
            tail.next = tail;//循环自身，形成循环
        } else {
            newNode.next = tail.next;// 新节点指向头节点
            tail.next = newNode;
        }
    }

    /**
     * 在指定的位置插入节点
     */
    public void insertAtPosition(int value, int position) {
        if (position <= 1) {
            insertAtHead(value);
            return;
        }

        Node newNode = new Node(value);
        Node current = tail.next; // 从头节点开始遍历
        int count = 1;

        while (count < position - 1 && current.next != tail.next) {
            current = current.next;
            count++;
        }

        newNode.next = current.next; // 新节点指向下一个节点
        current.next = newNode; // 当前节点指向新节点

        if (current == tail) { // 更新尾节点
            tail = newNode;
        }
    }

    /**
     * 删除指定值的节点
     *
     * @param value
     * @return
     */
    // 删除指定值的节点
    public boolean delete(int value) {
        if (tail == null) return false; // 链表为空

        Node current = tail.next; // 从头节点开始遍历
        Node prev = tail;

        do {
            if (current.value == value) {
                if (current == tail) { // 删除尾节点
                    if (current.next == current) { // 只有一个节点
                        tail = null;
                    } else {
                        prev.next = current.next;
                        tail = prev; // 更新尾节点
                    }
                } else {
                    prev.next = current.next; // 删除当前节点
                }
                return true; // 成功删除节点
            }
            prev = current;
            current = current.next;
        } while (current != tail.next);

        return false; // 未找到节点
    }

    // 遍历循环链表
    public void traverse() {
        if (tail == null) return; // 链表为空

        Node current = tail.next; // 从头节点开始
        do {
            System.out.print(current.value + " ");
            current = current.next;
        } while (current != tail.next);
        System.out.println();
    }
}
```
